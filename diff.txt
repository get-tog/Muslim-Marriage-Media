  <script>
    // Sample data (in real site, fetch from server)
    const sampleProfiles = [
      {id:1,name:'আহমাদ করিম',age:28,city:'ঢাকা',education:'স্নাতক',bio:'ইসলামিক শিক্ষায় আগ্রহী, পরিবার-কেন্দ্রিক।',photo:'https://i.pravatar.cc/150?img=3',gender:'male'},
      {id:2,name:'ফাতেমা হক',age:24,city:'চট্টগ্রাম',education:'মাস্টার্স',bio:'জব: শিক্ষক, নম্র ও বিনম্র স্বভাব।',photo:'https://i.pravatar.cc/150?img=5',gender:'female'},
      {id:3,name:'মুহিব করিম',age:30,city:'রাজশাহী',education:'ডিগ্রী',bio:'প্রকৌশলী, পরিবার-ভিত্তিক সম্পর্ক খুঁজছি।',photo:'https://i.pravatar.cc/150?img=8',gender:'male'},
      {id:4,name:'সাবরা জান',age:27,city:'সিলেট',education:'স্নাতক',bio:'গৃহশিক্ষিকা, ধর্মীয় কার্যক্রমে সক্রিয়।',photo:'https://i.pravatar.cc/150?img=12',gender:'female'}
    ];

    const profilesGrid = document.getElementById('profilesGrid');
    const modal = document.getElementById('modal');
    const modalContent = document.getElementById('modalContent');

    function renderProfiles(list){
      profilesGrid.innerHTML = '';
      list.forEach(p => {
        const card = document.createElement('div');
        card.className = 'bg-white p-4 rounded shadow flex gap-3 items-center';
        card.innerHTML = `
          <img src="${p.photo}" class="avatar" alt="${p.name}" />
          <div class="flex-1">
            <div class="flex items-center justify-between">
              <div>
                <h4 class="font-bold">${p.name}</h4>
                <p class="text-sm text-gray-600">${p.age} বছর • ${p.city}</p>
              </div>
              <div>
                <button class="viewBtn border px-3 py-1 rounded text-sm" data-id="${p.id}">ভিউ</button>
              </div>
            </div>
            <p class="text-sm mt-2 text-gray-700">${p.bio}</p>
          </div>
        `;
        profilesGrid.appendChild(card);
      });

      document.querySelectorAll('.viewBtn').forEach(btn => {
        btn.addEventListener('click', (e)=>{
          const id = +e.target.dataset.id;
          const user = sampleProfiles.find(x=>x.id===id);
          openProfileModal(user);
        });
      });
    }

    function openProfileModal(user){
      modalContent.innerHTML = `
        <div class="flex items-start gap-4">
          <img src="${user.photo}" class="w-24 h-24 rounded" />
          <div>
            <h3 class="text-xl font-bold">${user.name}</h3>
            <p class="text-sm text-gray-600">${user.age} বছর • ${user.city} • ${user.education}</p>
            <p class="mt-2">${user.bio}</p>
            <div class="mt-4 flex gap-2">
              <button id="contactBtn" class="bg-green-600 text-white px-3 py-1 rounded">যোগাযোগ করুন</button>
              <button id="closeModal" class="border px-3 py-1 rounded">বন্ধ</button>
            </div>
          </div>
        </div>
      `;
      modal.classList.remove('hidden');

      document.getElementById('closeModal').addEventListener('click', ()=> modal.classList.add('hidden'));
      document.getElementById('contactBtn').addEventListener('click', ()=>{
        modalContent.innerHTML += '<p class="mt-3 text-sm text-gray-600">যোগাযোগের জন্য প্রথমে লগইন/রেজিস্টার করুন।</p>';
      });
    }

    // Search
    document.getElementById('doSearch').addEventListener('click', ()=>{
      const gender = document.getElementById('gender').value;
      const min = parseInt(document.getElementById('ageMin').value||18,10);
      const max = parseInt(document.getElementById('ageMax').value||65,10);
      const city = document.getElementById('city').value.trim().toLowerCase();

      const filtered = sampleProfiles.filter(p=>{
        if(gender!=='any' && p.gender!==gender) return false;
        if(p.age < min || p.age > max) return false;
        if(city && !p.city.toLowerCase().includes(city)) return false;
        return true;
      });
      renderProfiles(filtered);
    });

    document.getElementById('resetSearch').addEventListener('click', ()=>{
      document.getElementById('gender').value='any';
      document.getElementById('ageMin').value='18';
      document.getElementById('ageMax').value='45';
      document.getElementById('city').value='';
      renderProfiles(sampleProfiles);
    });

    // Register modal (simple client-side demo)
    function openRegisterModal(){
      modalContent.innerHTML = `
        <h3 class="text-xl font-bold mb-2">নতুন প্রোফাইল রেজিস্টার</h3>
        <form id="registerForm" class="space-y-2">
          <input required name="name" placeholder="সম্পূর্ণ নাম" class="w-full border px-2 py-2 rounded" />
          <div class="flex gap-2">
            <input required name="age" type="number" placeholder="বয়স" class="w-1/2 border px-2 py-2 rounded" />
            <input name="city" placeholder="শহর" class="w-1/2 border px-2 py-2 rounded" />
          </div>
          <input name="education" placeholder="শিক্ষাগত যোগ্যতা" class="w-full border px-2 py-2 rounded" />
          <select name="gender" class="w-full border px-2 py-2 rounded">
            <option value="male">পুরুষ</option>
            <option value="female">নারী</option>
          </select>
          <textarea name="bio" placeholder="স্ব-মোটা পরিচিতি" class="w-full border px-2 py-2 rounded"></textarea>
          <div class="flex gap-2">
            <button type="submit" class="bg-green-600 text-white px-3 py-2 rounded">সাবমিট</button>
            <button type="button" id="closeReg" class="border px-3 py-2 rounded">বন্ধ</button>
          </div>
        </form>
      `;
      modal.classList.remove('hidden');

      document.getElementById('closeReg').addEventListener('click', ()=> modal.classList.add('hidden'));
      document.getElementById('registerForm').addEventListener('submit', (e)=>{
        e.preventDefault();
        const fd = new FormData(e.target);
        const newP = {
          id: Date.now(),
          name: fd.get('name'),
          age: +fd.get('age'),
          city: fd.get('city')||'---',
          education: fd.get('education')||'---',
          bio: fd.get('bio')||'---',
          photo: 'https://i.pravatar.cc/150?img='+Math.floor(Math.random()*70),
          gender: fd.get('gender')
        };
        sampleProfiles.push(newP);
        renderProfiles(sampleProfiles);
        modal.classList.add('hidden');
        alert('আপনার প্রোফাইল ডেমোভাবে যোগ করা হয়েছে। (রিয়েল সাইটে সার্ভার-সাইড রেজিস্ট্রেশন প্রয়োজন)');
      });
    }

    document.getElementById('openRegister').addEventListener('click', openRegisterModal);
    document.getElementById('quickRegister').addEventListener('click', (e)=>{ e.preventDefault(); openRegisterModal(); });
    document.getElementById('openSearch').addEventListener('click', ()=>{ window.location.hash='profiles'; });

    // Close modal on bg click
    modal.addEventListener('click', (e)=>{ if(e.target===modal) modal.classList.add('hidden'); });

    // initial render
    renderProfiles(sampleProfiles);
  </script>
